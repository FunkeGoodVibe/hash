def greet_in_yoruba(name):
    greetings = {
        "Good morning": "E kaaro",
        "Good afternoon": "E kaasan",
        "Good evening": "E kaale",
        "Welcome": "E kaabo",
    }
    
    print(f"{greetings['Welcome']}, {name}!")

# Example usage
greet_in_yoruba("Ayo")

# This script simulates a simple conversation in Yoruba

class YorubaConversation:
    
    def __init__(self):
        self.conversations = [
            {"speaker": "A", "message": "Bawo ni o se wa?"},
            {"speaker": "B", "message": "Mo wa dada, ese. Bawo ni e tile wa?"},
            {"speaker": "A", "message": "A wa daadaa pelu, a dupẹ."},
            {"speaker": "B", "message": "Kini iroyin tuntun?"},
            {"speaker": "A", "message": "Ko si gbogbo nkan lo wa n'ile, se n o le ran eleti kuro?"},
            {"speaker": "B", "message": "Lai lai! Inu mi wa dun lati ri ọ!"},
            {"speaker": "A", "message": "Inu mi wa dun gbe, ẹ s'ọ pọ lori ile ya."},
            {"speaker": "B", "message": "Ṣe dandan lati gbọ idẹ ni ana."},
            {"speaker": "A", "message": "Ko si wahala, a maa gbọ."},
            {"speaker": "B", "message": "Akiyesi daadaa. tun iru?", "b"): "OH!",
        ]

    def display(self):
        for line in self.conversations:
            print(f'{line["speaker"]}: {line["message"]}')

# Instantiate and display conversation
conversation = YorubaConversation()
